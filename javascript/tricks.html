# 技巧

  • 插件库

    The Vanilla JS Toolkit (opens new window)

  • 建议不要使用||来接默认值，因为这样，就不能赋值为 0 了，0 会被当作假值而取默认值。

    function setAge(age) {
  this.age = age || 10
  console.log(this.age)
}
setAge(0) // 10

    改进后的代码，只有在 age 为 undefined 时才取默认值。

    function setAge(age) {
  this.age = typeof age !== 'undefined' ? age : 10
  console.log(this.age)
}
setAge(0) // 0

    typeof的目的是确保不会抛出ReferenceError（引用错误）

  • bind 的妙用

;['foo', 'bar', 'baz'].filter(RegExp.prototype.test.bind(/a/)) // ["bar", "baz"]
  • 用+号把字符串转成数值
var string = '500'
var num = +string
  • 用!!将值强制转为布尔值
!!null // false
!!undefined // false
!!'false' // true :(
!!'' // false
  • sort 方法对数字排序
;[10, 7, 9, 8, 11].sort(function(a, b) {
  return a - b
})
// => [7, 8, 9, 10, 11]
  • 返回毫秒
;+new Date() // 1259359833574
  • 数组追加
var a = [4, 5, 6]
var b = [7, 8]

Array.prototype.push.apply(a, b)
console.log(a)
  • 位数补齐
function PrefixInteger(num, length) {
  return (Array(length).join('0') + num).slice(-length)
}

PrefixInteger(233, 4) // "0233"
  • 创建指定长度随机字符串
function RandomString(length) {
  var str = ''
  for (
    ;
    str.length < length;
    str += Math.random()
      .toString(36)
      .substr(2)
  );
  return str.substr(0, length)
}

console.log(RandomString(23))

# 从 Object 构建 URL querystring

// build a query string from an object
function buildQuery(data) {
  if (typeof data === 'string') return data

  const query = []

  for (const key in data) {
    if (data.hasOwnProperty(key)) {
      query.push(encodeURIComponent(key) + '=' + encodeURIComponent(data[key]))
    }
  }

  return query.join('&')
}
